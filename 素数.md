**枚举速度较慢**
```c++
bool isPrime(int n)
{
	if (n == 1) return false;
	int sqr = sqrt(1.0*n);
	for (int i = 2; i <= sqr; i++)
	{
		if (n%i == 0) return false;
	}
	return true;
}
```

**筛选法速度较快**

```c++
#include<iostream>
#include<string>
#include<vector>
#include<algorithm>
using namespace std;
const int maxn = 1000000;
bool p[maxn];//false 为素数
int main()
{
	vector<int> primeList,ans;
	p[0] = p[1] = true;
	for (int i = 2; i < maxn; i++)
	{
		if (p[i] == false)
		{
			primeList.push_back(i);
			for (int j = i + i; j < maxn; j += i)
				p[j] = true;
		}
	}
         return 0;
}
```
